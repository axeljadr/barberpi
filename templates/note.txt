CREATE TABLE Citas (
    id_cita int primary key identity,
    id_usuario INT NOT NULL,
    id_barbero INT NOT NULL,
    fecha DATE NOT NULL,
    hora TIME NOT NULL,
    estado varchar(14) DEFAULT 'pendiente',
    notas TEXT,
    precio_estimado DECIMAL(5,2),
    CHECK (estado IN ('pendiente', 'confirmada', 'cancelada', 'completada')),
    FOREIGN KEY (id_usuario) REFERENCES Usuarios(id_usuario),
    FOREIGN KEY (id_barbero) REFERENCES usuarios(id_usuario)
);  

id_cita, id_usuario, id_barbero, fecha, hora, estado, notas, precio_estimado
